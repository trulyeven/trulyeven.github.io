      const videoCode = matchResult ? matchResult[1] : '';
      if (videoCode === null || videoCode === '') {
        submitButton.disabled = true;
        inputalert.hidden = false;
    } else {
        submitButton.disabled = false;
        inputalert.hidden = true;
      }
